# HIDE CODE


thres_center = 10
thres_opticflow = 1e-1
df = []
for name in file:
    print(name)
    group = file[name]
    nx = np.array(group['nx']).item()
    ny = np.array(group['ny']).item()
    sres = np.unique(group['spatres']).item()
    assert nx == ny 

    hf = np.array(group['hyperflow'])
    hf = HyperFlow(
        opticflow=hf[:, 2:],
        center=hf[:, :2],
        size=nx,
        sres=sres,
        radius=7,
    )
    cutoff_center = []
    cutoff_opticflow = []
    f, px = hf.show_psd('center', False)
    for x in px:
        max_i = np.max(np.where(x > thres_center)[0])
        cutoff_center.append(np.round(f[max_i], 2))
    f, px = hf.show_psd('opticflow', False, cutoff=5)
    for x in px:
        max_i = np.max(np.where(x > thres_opticflow)[0])
        cutoff_opticflow.append(np.round(f[max_i], 2))

    df.append({
        'name': [name] * 2,
        'label': ['center'] * 2,
        'idx': range(2),
        'sd': hf.center.std(0),
        'mu': hf.center.mean(0),
        'cutoff': cutoff_center,
    })
    df.append({
        'name': [name] * 6,
        'label': ['opticflow'] * 6,
        'idx': range(6),
        'sd': hf.opticflow.std(0),
        'mu': hf.opticflow.mean(0),
        'cutoff': cutoff_opticflow,
    })
    msg = ', '.join([
        name,
        f"cutoffs (center): {cutoff_center}",
        f"cutoffs (opticflow): {cutoff_opticflow}",
    ])
    print(msg)
    print('_' * 70)
    print('\n')

df = pd.DataFrame(merge_dicts(df))
print('Done.')
file.close()
